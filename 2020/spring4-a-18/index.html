FactoryBean创建bean

Written by Bing with ♥ on June 6, 2020 in spring

Java

package org.ouhei.spring.factory;

import org.ouhei.spring.beans.Student;
import org.springframework.beans.factory.FactoryBean;

/**
 * @author OuHei
 * @create 2020/06/07
 */
public class DemoFactoryBean implements FactoryBean<Student> {

    public Student getObject() throws Exception {
        return new Student("ouhei",13014,91.1);
    }

    public Class<?> getObjectType() {
        return Student.class;
    }

    public boolean isSingleton() {
        return true;
    }
}

xml

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
       xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd">
    <!--
        通过FactoryBean来配置Bean的实例
        class：指向FactoryBean的全类名
        property配置FactoryBean的属性

        返回的实例是FactoryBean的getObject()方法返回的实例
    -->
    <bean id="student" class="org.ouhei.spring.factory.DemoFactoryBean">

    </bean>
</beans>

TestMain

package org.ouhei.spring.main;

import org.ouhei.spring.beans.Student;
import org.springframework.context.ApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

/**
 * @author OuHei
 * @create 2020/02/28
 */
public class TestMain {
    public static void main(String[] args) {

        ApplicationContext ctx = new ClassPathXmlApplicationContext("bean-factory.xml");
        Student student = (Student)ctx.getBean("student");
        System.out.println(student);
    }
}
