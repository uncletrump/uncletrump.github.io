集合属性(map)

Java.util.Map 通过 标签定义, 标签里可以使用多个 作为子标签. 每个条目包含一个键和一个值.

必须在 标签里定义键

因为键和值的类型没有限制, 所以可以自由地为它们指定 , , 或 元素.

可以将 Map 的键和值作为 的属性定义: 简单常量使用 key 和 value 来定义; Bean 引用通过 key-ref 和 value-ref 属性定义.

使用 定义 java.util.Properties, 该标签使用多个 作为子标签. 每个 标签必须定义 key 属性.

JavaBean

package org.ouhei.spring.collections;

import java.util.Map;

/**
 * @author OuHei
 * @create 2020/03/01
 */
public class Teacher {

    private String name;

    private Integer age;

    private Map<String,Student> student;

    public Teacher() {
    }

    public Teacher(String name, Integer age, Map<String, Student> student) {
        this.name = name;
        this.age = age;
        this.student = student;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public Integer getAge() {
        return age;
    }

    public void setAge(Integer age) {
        this.age = age;
    }

    public Map<String, Student> getStudent() {
        return student;
    }

    public void setStudent(Map<String, Student> student) {
        this.student = student;
    }

    @Override
    public String toString() {
        return "Teacher{" +
                "name='" + name + '\'' +
                ", age=" + age +
                ", student=" + student +
                '}';
    }
}


package org.ouhei.spring.main;

import org.ouhei.spring.beans.HelloWord;
import org.ouhei.spring.collections.Teacher;
import org.springframework.context.ApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

/**
 * @author OuHei
 * @create 2020/02/28
 */
public class TestMain {
    public static void main(String[] args) {

        // 创建Spring IOC容器，并会调用构造方法，通过set方法对属性进行赋值
        ApplicationContext ac=new ClassPathXmlApplicationContext("applicationContext.xml");

        // 从容器中获取bean对象
        Teacher teacher = (Teacher) ac.getBean("teacher");
        System.out.println(teacher.getStudent().get("1"));
        System.out.println(teacher.getStudent().get("2"));



    }
}

applicationContext.xml

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
       xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd">

    <!--类型和索引可以混合使用-->
    <!--按类型匹配-->
    <bean id="student1" class="org.ouhei.spring.collections.Student">
        <constructor-arg value="小明" type="java.lang.String"/>
        <constructor-arg value="13014" type="java.lang.Integer"/>
        <constructor-arg value="521.34" type="java.lang.Double"/>
    </bean>

    <!--按索引配置-->
    <!--可以用子标签value进行赋值，输出特殊字符时<![CDATA[]]>-->
    <bean id="student2" class="org.ouhei.spring.collections.Student">
        <constructor-arg index="0">
            <value><![CDATA[<小红>]]></value>
        </constructor-arg>
        <constructor-arg value="13015" index="1"/>
        <constructor-arg type="java.lang.Double">
            <value>692.34</value>
        </constructor-arg>
    </bean>

    <bean id="teacher" class="org.ouhei.spring.collections.Teacher">
        <property name="name" value="波多野结衣"/>
        <property name="age" value="30"/>
        <property name="student">
            <map>
                <entry key="1" value-ref="student1"/>
                <entry key="2" value-ref="student2"/>
            </map>
        </property>
    </bean>

</beans>

模拟获取properties文件的值

JavaBean

public class DataSource {
    private Properties properties;

    public DataSource() {
    }

    public DataSource(Properties properties) {
        this.properties = properties;
    }

    public Properties getProperties() {
        return properties;
    }

    public void setProperties(Properties properties) {
        this.properties = properties;
    }

    @Override
    public String toString() {
        return "DataSource{" +
                "properties=" + properties +
                '}';
    }
}


public class TestMain {
    public static void main(String[] args) {

        // 创建Spring IOC容器，并会调用构造方法，通过set方法对属性进行赋值
        ApplicationContext ac=new ClassPathXmlApplicationContext("applicationContext.xml");

        // 从容器中获取bean对象
        DataSource dataSource = (DataSource) ac.getBean("dataSource");
        System.out.println(dataSource.getProperties().getProperty("user"));
        System.out.println(dataSource.getProperties().getProperty("password"));
        System.out.println(dataSource.getProperties().getProperty("jdbcUrl"));
        System.out.println(dataSource.getProperties().getProperty("driverClass"));

    }
}

applicationContext.xml

<bean id="dataSource" class="org.ouhei.spring.collections.DataSource">
        <property name="properties">
            <props>
                <prop key="user">root</prop>
                <prop key="password">root</prop>
                <prop key="jdbcUrl">jdbc</prop>
                <prop key="driverClass">jdbc.driver</prop>
            </props>
        </property>
    </bean>
