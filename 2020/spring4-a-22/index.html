基于注解的方式

  1. 配置文件中加入如下配置

       
<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
      xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
      xmlns:context="http://www.springframework.org/schema/context"
      xmlns:aop="http://www.springframework.org/schema/aop"
      xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd http://www.springframework.org/schema/context https://www.springframework.org/schema/context/spring-context.xsd http://www.springframework.org/schema/aop http://www.springframework.org/schema/aop/spring-aop.xsd">
   
   <!--注解 扫描包-->
   <context:component-scan base-package="org.ouhei.spring">
   </context:component-scan>
   <!--AOP注解-->
   <aop:aspectj-autoproxy></aop:aspectj-autoproxy>
   
</beans>

  2. 把横切关注点的代码抽象到切面的泪中

    • 切面首先是一个IOC中的bean，即加入@Component注解
    • 切面还要加入@Aspect注解

  3. 在类中声明各种通知

    • 声明一个方法

    • 在方法前加入通知注解

      package org.ouhei.spring.aop;
   
import org.aspectj.lang.JoinPoint;
import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.After;
import org.aspectj.lang.annotation.AfterReturning;
import org.aspectj.lang.annotation.AfterThrowing;
import org.aspectj.lang.annotation.Around;
import org.aspectj.lang.annotation.Aspect;
import org.aspectj.lang.annotation.Before;
import org.springframework.stereotype.Component;
   
import java.util.Arrays;
   
@Component
@Aspect
public class Demo {
   
// 前置通知:在方法执行之前执行的通知
@Before("execution(public int org.ouhei.spring.serviceImpl.TestServiceImpl.sum(int,int))")
public void beforeTest(JoinPoint joinPoint) {
   String methodName = joinPoint.getSignature().getName();
   Object[] args = joinPoint.getArgs();
   System.err.println("the method " + methodName + " begin with " + Arrays.asList(args));
}
   
// 后置通知是在方法执行之后执行的通知(无论是否发生异常)
@After("execution(* org.ouhei.spring.serviceImpl.*.*(int,int))")
public void afterTest(JoinPoint joinPoint) {
   String methodName = joinPoint.getSignature().getName();
   Object[] args = joinPoint.getArgs();
   System.err.println("the method " + methodName + " end with " + Arrays.asList(args));
}
   
// 返回通知是在方法执行之后，有正常返回值的情况下执行的通知(发生异常时，没有正常返回值时，则不通知)
@AfterReturning(value = "execution(* org.ouhei.spring.serviceImpl.*.*(int,int))", returning = "result")
public void afterReturningTest(JoinPoint joinPoint, Object result) {
   String methodName = joinPoint.getSignature().getName();
   Object[] args = joinPoint.getArgs();
   System.err.println("the method " + methodName + " end with " + Arrays.asList(args) + " result " + result);
}
   
// 异常通知是在抛出异常时执行的通知（可以只在特定的异常时通知）
@AfterThrowing(value = "execution(* org.ouhei.spring.serviceImpl.*.*(int,int))", throwing = "ex")
public void afterThrowingTest(JoinPoint joinPoint, Exception ex) {
   String methodName = joinPoint.getSignature().getName();
   Object[] args = joinPoint.getArgs();
   System.err.println("the method " + methodName + " end with " + Arrays.asList(args) + " Exception " + ex);
}
   
// 环绕通知需要携带ProceedingJoinPoint类型的参数
// 环绕通知类似于动态代理全过程：ProceedingJoinPoint类型的参数可以决定是否执行目标方法
// 环绕通知必须有返回值，返回值必为目标方法的返回值
@Around(value = "execution(* org.ouhei.spring.serviceImpl.*.*(int,int))")
public Object aroundTest(ProceedingJoinPoint proceedingJoinPoint) {
   Object result = null;
   String methodName = proceedingJoinPoint.getSignature().getName();
   try {
       // 此处即是执行目标方法
       // 可以在之前或者之后模拟前置和后置通知
       result = proceedingJoinPoint.proceed();
   } catch (Throwable e) {
       // 可以在此模拟异常通知
       throw new RuntimeException(e);
   }
   return result;
}
}

       ApplicationContext ctx = new ClassPathXmlApplicationContext("bean-factory.xml");

   }
