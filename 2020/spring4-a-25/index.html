XML配置AOP

Written by Bing with ♥ on July 11, 2020 in spring

xml

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
       xmlns:context="http://www.springframework.org/schema/context"
       xmlns:aop="http://www.springframework.org/schema/aop"
       xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd http://www.springframework.org/schema/context https://www.springframework.org/schema/context/spring-context.xsd http://www.springframework.org/schema/aop http://www.springframework.org/schema/aop/spring-aop.xsd">

    <!--配置bean-->
    <bean id="testService"
          class="org.ouhei.spring.serviceImpl.TestServiceImpl"></bean>

    <!--配置切面-->
    <bean id="demo" class="org.ouhei.spring.aop.Demo"></bean>

    <!--配置AOP-->
    <aop:config>
        <!--配置切点表达式-->
        <aop:pointcut id="pointcut" expression="execution(public int org.ouhei.spring.serviceImpl.TestServiceImpl.sum(int,int))"/>
        <!--配置切面及通知-->
        <aop:aspect ref="demo" order="0">
            <aop:before method="beforeTest" pointcut-ref="pointcut"></aop:before>
            <aop:after method="afterTest" pointcut-ref="pointcut"></aop:after>
        </aop:aspect>
        <!--配置切面及通知-->
        <!--配置切面及通知-->
    </aop:config>
</beans>

切面

package org.ouhei.spring.aop;

import org.aspectj.lang.JoinPoint;
import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.After;
import org.aspectj.lang.annotation.AfterReturning;
import org.aspectj.lang.annotation.AfterThrowing;
import org.aspectj.lang.annotation.Around;
import org.aspectj.lang.annotation.Aspect;
import org.aspectj.lang.annotation.Before;
import org.springframework.stereotype.Component;

import java.util.Arrays;

/**
 * @author OuHei
 * @create 2020/06/18
 */
public class Demo {

    public void beforeTest(JoinPoint joinPoint) {
        String methodName = joinPoint.getSignature().getName();
        Object[] args = joinPoint.getArgs();
        System.err.println("the method " + methodName + " begin with " + Arrays.asList(args));
    }

    public void afterTest(JoinPoint joinPoint) {
        String methodName = joinPoint.getSignature().getName();
        Object[] args = joinPoint.getArgs();
        System.err.println("the method " + methodName + " end with " + Arrays.asList(args));
    }

    public void afterReturningTest(JoinPoint joinPoint, Object result) {
        String methodName = joinPoint.getSignature().getName();
        Object[] args = joinPoint.getArgs();
        System.err.println("the method " + methodName + " end with " + Arrays.asList(args) + " result " + result);
    }

    public void afterThrowingTest(JoinPoint joinPoint, Exception ex) {
        String methodName = joinPoint.getSignature().getName();
        Object[] args = joinPoint.getArgs();
        System.err.println("the method " + methodName + " end with " + Arrays.asList(args) + " Exception " + ex);
    }

    public Object aroundTest(ProceedingJoinPoint proceedingJoinPoint) {
        Object result = null;
        String methodName = proceedingJoinPoint.getSignature().getName();
        try {
            // 此处即是执行目标方法
            // 可以在之前或者之后模拟前置和后置通知
            result = proceedingJoinPoint.proceed();
        } catch (Throwable e) {
            // 可以在此模拟异常通知
            throw new RuntimeException(e);
        }
        return result;
    }
}

package org.ouhei.spring.service;

/**
 * @author OuHei
 * @create 2020/06/23
 */
public interface TestService {
    int sum(int a,int b);
}



package org.ouhei.spring.serviceImpl;

import org.ouhei.spring.service.TestService;

/**
 * @author OuHei
 * @create 2020/06/23
 */
public class TestServiceImpl implements TestService {
    public int sum(int a, int b) {
        System.out.println(a + b);
        return a + b;
    }
}

运行

package org.ouhei.spring.main;

import org.ouhei.spring.service.TestService;
import org.springframework.context.ApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

/**
 * @author OuHei
 * @create 2020/02/28
 */
public class TestMain {

    public static void main(String[] args) {

        ApplicationContext ctx = new ClassPathXmlApplicationContext("bean-factory.xml");
        TestService bean = (TestService) ctx.getBean("testService");
        bean.sum(3, 5);
    }
}
