实例工厂方法创建bean

Written by Bing with ♥ on May 14, 2020 in spring

java

/**
 * @author OuHei
 * @create 2020/05/14
 */
public class InstanceFactory {
    private Map<String, Student> students;
    public InstanceFactory(){
        students = new HashMap<String, Student>();
        students.put("13010",new Student("小明",13010,85.0));
        students.put("13014",new Student("ouhei",13014,86.0));
    }

    public Student getStudent(String key){
        return students.get(key);
    }
}

TestMain

/**
 * @author OuHei
 * @create 2020/02/28
 */
public class TestMain {
    public static void main(String[] args) {

        ApplicationContext ctx = new ClassPathXmlApplicationContext("bean-factory.xml");
        Student student = (Student)ctx.getBean("student");
        System.out.println(student);
    }
}

xml

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
       xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd">
    <!--
        factory-bean:指向实例工厂方法的bean
        factory-method:指向静态工厂方法的名字
        果工厂方法需要传入参数，则使用constructor-arg来配置参数
    -->
    <bean id="factory" class="org.ouhei.spring.factory.InstanceFactory"></bean>
    <bean id="student" factory-bean="factory" factory-method="getStudent">

        <constructor-arg value="13014"/>
    </bean>
</beans>
