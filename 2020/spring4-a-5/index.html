集合属性(list)

在 Spring中可以通过一组内置的 xml 标签(例如: , 或 ) 来配置集合属性。

配置 java.util.List 类型的属性, 需要指定 标签, 在标签里包含一些元素. 这些标签可以通过 指定简单的常量值, 通过 指定对其他 Bean 的引用. 通过 指定内置 Bean 定义. 通过 指定空元素. 甚至可以内嵌其他集合。

数组的定义和 List 一样, 都使用 。

配置 java.util.Set 需要使用 标签, 定义元素的方法与 List 一样。

JavaBean

package org.ouhei.spring.collections;

import java.util.List;

/**
 * @author OuHei
 * @create 2020/03/01
 */
public class Teacher {

    private String name;

    private Integer age;

    private List<Student> student;

    public Teacher() {
    }

    public Teacher(String name, Integer age, List<Student> student) {
        this.name = name;
        this.age = age;
        this.student = student;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public Integer getAge() {
        return age;
    }

    public void setAge(Integer age) {
        this.age = age;
    }

    public List<Student> getStudent() {
        return student;
    }

    public void setStudent(List<Student> student) {
        this.student = student;
    }

    @Override
    public String toString() {
        return "Teacher{" +
                "name='" + name + '\'' +
                ", age=" + age +
                ", student=" + student +
                '}';
    }
}

applicationContext.xml

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
       xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd">

    <!--类型和索引可以混合使用-->
    <!--按类型匹配-->
    <bean id="student1" class="org.ouhei.spring.collections.Student">
        <constructor-arg value="小明" type="java.lang.String"/>
        <constructor-arg value="13014" type="java.lang.Integer"/>
        <constructor-arg value="521.34" type="java.lang.Double"/>
    </bean>

    <!--按索引配置-->
    <!--可以用子标签value进行赋值，输出特殊字符时<![CDATA[]]>-->
    <bean id="student2" class="org.ouhei.spring.collections.Student">
        <constructor-arg index="0">
            <value><![CDATA[<小红>]]></value>
        </constructor-arg>
        <constructor-arg value="13015" index="1"/>
        <constructor-arg type="java.lang.Double">
            <value>692.34</value>
        </constructor-arg>
    </bean>

    <bean id="teacher" class="org.ouhei.spring.collections.Teacher">
        <property name="name" value="波多野结衣"/>
        <property name="age" value="30"/>
        <property name="student">
            <list>
                <ref bean="student1"/>
                <ref bean="student2"/>
                <bean class="org.ouhei.spring.collections.Student">
                    <property name="name" value="小黑"/>
                    <property name="num" value="13017"/>
                    <property name="point" value="690"/>
                </bean>
            </list>
        </property>
    </bean>

</beans>
